static void yar_runtime_fail(const char *data, long long len) {
    if (len > 0) {
        (void)fwrite(data, 1, (size_t)len, stderr);
    }
    (void)fflush(stderr);
    exit(1);
}

void yar_print(const char *data, long long len) {
    if (len <= 0) {
        return;
    }
    (void)fwrite(data, 1, (size_t)len, stdout);
}

void yar_panic(const char *data, long long len) {
    yar_runtime_fail(data, len);
}

void yar_eprint(const char *data, long long len) {
    if (len <= 0) {
        return;
    }
    (void)fwrite(data, 1, (size_t)len, stderr);
    (void)fflush(stderr);
}

void yar_trap_oom(void) {
    static const char message[] = "runtime failure: out of memory\n";
    yar_runtime_fail(message, sizeof(message) - 1);
}

typedef struct yar_gc_block {
    struct yar_gc_block *next;
    size_t size;
    int marked;
} yar_gc_block;

static yar_gc_block *yar_gc_blocks = NULL;
static size_t yar_gc_bytes = 0;
static size_t yar_gc_heap_target = 1024 * 1024;
static int yar_gc_configured = 0;
static int yar_gc_collecting = 0;
static void *yar_gc_stack_top = NULL;

static void yar_gc_mark_range(const void *start, const void *end);
static void yar_gc_mark_pointer(const void *candidate);
static void yar_gc_mark_block(yar_gc_block *block);
static void yar_gc_grow_target(size_t min_live_bytes);

static void yar_gc_configure(void) {
    const char *value;
    char *end;
    unsigned long long parsed;

    if (yar_gc_configured) {
        return;
    }
    yar_gc_configured = 1;

    value = getenv("YAR_GC_HEAP_TARGET_BYTES");
    if (value == NULL || value[0] == '\0') {
        return;
    }

    errno = 0;
    parsed = strtoull(value, &end, 10);
    if (errno != 0 || end == value || *end != '\0' || parsed == 0) {
        return;
    }
    yar_gc_heap_target = (size_t)parsed;
}

static void *yar_gc_data(yar_gc_block *block) {
    return (void *)(block + 1);
}

static const void *yar_gc_const_data(const yar_gc_block *block) {
    return (const void *)(block + 1);
}

static yar_gc_block *yar_gc_find_block(const void *candidate) {
    yar_gc_block *block;

    if (candidate == NULL) {
        return NULL;
    }

    for (block = yar_gc_blocks; block != NULL; block = block->next) {
        const unsigned char *start = (const unsigned char *)yar_gc_const_data(block);
        const unsigned char *end = start + block->size;
        const unsigned char *ptr = (const unsigned char *)candidate;
        if (ptr >= start && ptr < end) {
            return block;
        }
    }
    return NULL;
}

static int yar_gc_owns_pointer(const void *candidate) {
    return yar_gc_find_block(candidate) != NULL;
}

static void yar_gc_mark_block(yar_gc_block *block) {
    if (block == NULL || block->marked) {
        return;
    }
    block->marked = 1;
    yar_gc_mark_range(yar_gc_const_data(block), ((const unsigned char *)yar_gc_const_data(block)) + block->size);
}

static void yar_gc_mark_pointer(const void *candidate) {
    yar_gc_mark_block(yar_gc_find_block(candidate));
}

static void yar_gc_mark_range(const void *start, const void *end) {
    uintptr_t lo;
    uintptr_t hi;
    uintptr_t cursor;

    if (start == NULL || end == NULL) {
        return;
    }

    lo = (uintptr_t)start;
    hi = (uintptr_t)end;
    if (lo > hi) {
        uintptr_t tmp = lo;
        lo = hi;
        hi = tmp;
    }
    if (hi - lo < sizeof(uintptr_t)) {
        return;
    }

    /*
     * Conservative scanning must stay within the actual byte range.
     * Short strings and small structs may not be pointer-aligned, so scan each
     * byte offset rather than rounding the range outward and risking OOB reads.
     */
    for (cursor = lo; cursor + sizeof(uintptr_t) <= hi; cursor++) {
        uintptr_t candidate = 0;
        memcpy(&candidate, (const void *)cursor, sizeof(candidate));
        if (candidate == 0) {
            continue;
        }
        yar_gc_mark_pointer((const void *)candidate);
    }
}

void yar_gc_init_stack_top(void *stack_top) {
    yar_gc_configure();
    yar_gc_stack_top = stack_top;
}

void yar_gc_collect(void) {
    jmp_buf regs;
    volatile int stack_marker = 0;
    yar_gc_block **link;

    yar_gc_configure();
    if (yar_gc_collecting) {
        return;
    }

    yar_gc_collecting = 1;
    if (setjmp(regs) == 0) {
        yar_gc_mark_range(&regs, ((const unsigned char *)&regs) + sizeof(regs));
        if (yar_gc_stack_top != NULL) {
            yar_gc_mark_range((const void *)&stack_marker, yar_gc_stack_top);
        }
    }

    link = &yar_gc_blocks;
    while (*link != NULL) {
        yar_gc_block *block = *link;
        if (!block->marked) {
            *link = block->next;
            yar_gc_bytes -= block->size;
            free(block);
            continue;
        }
        block->marked = 0;
        link = &block->next;
    }

    yar_gc_grow_target(yar_gc_bytes);
    yar_gc_collecting = 0;
}

static void yar_gc_grow_target(size_t min_live_bytes) {
    size_t next;

    if (yar_gc_heap_target == 0) {
        yar_gc_heap_target = 1024;
    }
    next = yar_gc_heap_target;
    if (min_live_bytes <= next) {
        return;
    }
    while (next < min_live_bytes) {
        if (next > ((size_t)-1) / 2) {
            next = min_live_bytes;
            break;
        }
        next *= 2;
    }
    yar_gc_heap_target = next;
}

static void yar_gc_maybe_collect(size_t incoming_size) {
    volatile int stack_marker = 0;
    size_t required = yar_gc_bytes + incoming_size;

    yar_gc_configure();
    if (yar_gc_stack_top == NULL) {
        yar_gc_stack_top = (void *)&stack_marker;
    }
    if (required <= yar_gc_heap_target || yar_gc_collecting) {
        return;
    }

    yar_gc_collect();
    required = yar_gc_bytes + incoming_size;
    if (required > yar_gc_heap_target) {
        size_t grow_to = required;
        if (yar_gc_bytes > 0 && grow_to < yar_gc_bytes * 2) {
            grow_to = yar_gc_bytes * 2;
        }
        yar_gc_grow_target(grow_to);
    }
}

static void *yar_gc_alloc_bytes(size_t size, int zeroed) {
    yar_gc_block *block;
    size_t total_size;

    if (size == 0) {
        size = 1;
    }

    yar_gc_maybe_collect(size);

    total_size = sizeof(yar_gc_block) + size;
    block = (yar_gc_block *)malloc(total_size);
    if (block == NULL) {
        yar_gc_collect();
        block = (yar_gc_block *)malloc(total_size);
        if (block == NULL) {
            yar_trap_oom();
        }
    }

    block->size = size;
    block->marked = 0;
    block->next = yar_gc_blocks;
    yar_gc_blocks = block;
    yar_gc_bytes += size;

    if (zeroed) {
        memset(yar_gc_data(block), 0, size);
    }
    return yar_gc_data(block);
}

void yar_slice_index_check(long long index, long long len) {
    if (index < 0 || index >= len) {
        static const char message[] = "runtime failure: slice index out of range\n";
        yar_runtime_fail(message, sizeof(message) - 1);
    }
}

void yar_slice_range_check(long long start, long long end, long long len) {
    if (start < 0 || end < start || end > len) {
        static const char message[] = "runtime failure: slice range out of bounds\n";
        yar_runtime_fail(message, sizeof(message) - 1);
    }
}

void *yar_alloc(long long size) {
    if (size < 0) {
        static const char message[] = "runtime failure: invalid allocation size\n";
        yar_runtime_fail(message, sizeof(message) - 1);
    }
    return yar_gc_alloc_bytes((size_t)size, 0);
}

static int32_t yar_host_argc = 0;
static char **yar_host_argv = NULL;

void yar_set_args(int32_t argc, char **argv) {
    if (argc < 0) {
        argc = 0;
    }
    yar_host_argc = argc;
    yar_host_argv = argv;
}

int32_t yar_str_equal(const char *a_ptr, long long a_len, const char *b_ptr, long long b_len) {
    if (a_len != b_len) return 0;
    if (a_len == 0) return 1;
    return memcmp(a_ptr, b_ptr, (size_t)a_len) == 0 ? 1 : 0;
}

yar_str yar_str_concat(const char *a_ptr, long long a_len, const char *b_ptr, long long b_len) {
    yar_str result;
    long long total = a_len + b_len;
    if (total == 0) {
        result.ptr = NULL;
        result.len = 0;
        return result;
    }
    char *buf = (char *)yar_alloc(total);
    if (a_len > 0) memcpy(buf, a_ptr, (size_t)a_len);
    if (b_len > 0) memcpy(buf + a_len, b_ptr, (size_t)b_len);
    result.ptr = buf;
    result.len = total;
    return result;
}

void yar_str_index_check(long long index, long long len) {
    if (index < 0 || index >= len) {
        static const char message[] = "runtime failure: string index out of range\n";
        yar_runtime_fail(message, sizeof(message) - 1);
    }
}

yar_str yar_str_from_byte(int32_t value) {
    yar_str result;
    if (value < 0 || value > 255) {
        static const char message[] = "runtime failure: chr value out of byte range\n";
        yar_runtime_fail(message, sizeof(message) - 1);
    }
    char *buf = (char *)yar_alloc(1);
    buf[0] = (char)(unsigned char)value;
    result.ptr = buf;
    result.len = 1;
    return result;
}

yar_str yar_to_str_i32(int32_t value) {
    char tmp[12];
    int neg = 0;
    int32_t v = value;
    if (v < 0) {
        neg = 1;
        v = -v;
        if (v < 0) {
            static const char m[] = "-2147483648";
            char *b = (char *)yar_alloc(11);
            for (int i = 0; i < 11; i++) b[i] = m[i];
            yar_str r; r.ptr = b; r.len = 11;
            return r;
        }
    }
    int pos = 11;
    if (v == 0) { tmp[--pos] = '0'; }
    else { while (v > 0) { tmp[--pos] = '0' + (char)(v % 10); v /= 10; } }
    if (neg) tmp[--pos] = '-';
    int len = 11 - pos;
    char *buf = (char *)yar_alloc(len);
    for (int i = 0; i < len; i++) buf[i] = tmp[pos + i];
    yar_str result; result.ptr = buf; result.len = len;
    return result;
}

yar_str yar_to_str_i64(int64_t value) {
    char tmp[21];
    int neg = 0;
    int64_t v = value;
    if (v < 0) {
        neg = 1;
        v = -v;
        if (v < 0) {
            static const char m[] = "-9223372036854775808";
            char *b = (char *)yar_alloc(20);
            for (int i = 0; i < 20; i++) b[i] = m[i];
            yar_str r; r.ptr = b; r.len = 20;
            return r;
        }
    }
    int pos = 20;
    if (v == 0) { tmp[--pos] = '0'; }
    else { while (v > 0) { tmp[--pos] = '0' + (char)(v % 10); v /= 10; } }
    if (neg) tmp[--pos] = '-';
    int len = 20 - pos;
    char *buf = (char *)yar_alloc(len);
    for (int i = 0; i < len; i++) buf[i] = tmp[pos + i];
    yar_str result; result.ptr = buf; result.len = len;
    return result;
}

/* --- host process and environment runtime --- */

#define YAR_HOST_OK                0
#define YAR_HOST_NOT_FOUND         1
#define YAR_HOST_PERMISSION_DENIED 2
#define YAR_HOST_INVALID_ARGUMENT  3
#define YAR_HOST_IO                4

typedef struct {
    int32_t exit_code;
    yar_str stdout;
    yar_str stderr;
} yar_process_result;

#ifdef _WIN32
static int32_t yar_host_status_from_lasterror(DWORD err) {
    switch (err) {
    case ERROR_FILE_NOT_FOUND:
    case ERROR_PATH_NOT_FOUND:
        return YAR_HOST_NOT_FOUND;
    case ERROR_ACCESS_DENIED:
        return YAR_HOST_PERMISSION_DENIED;
    case ERROR_BAD_ARGUMENTS:
    case ERROR_INVALID_PARAMETER:
        return YAR_HOST_INVALID_ARGUMENT;
    default:
        return YAR_HOST_IO;
    }
}
#else
static int32_t yar_host_status_from_errno(int errnum) {
    switch (errnum) {
    case 0:
        return YAR_HOST_IO;
    case ENOENT:
    case ENOTDIR:
        return YAR_HOST_NOT_FOUND;
    case EACCES:
    case EPERM:
        return YAR_HOST_PERMISSION_DENIED;
    case E2BIG:
    case EFAULT:
    case EINVAL:
    case ENAMETOOLONG:
    case ENOEXEC:
        return YAR_HOST_INVALID_ARGUMENT;
    default:
        return YAR_HOST_IO;
    }
}
#endif

static void yar_free_str(yar_str value) {
    if (value.ptr == NULL) {
        return;
    }
    if (yar_gc_owns_pointer(value.ptr)) {
        return;
    }
    free((void *)value.ptr);
}

static int32_t yar_host_string_cstr(yar_str value, int allow_empty, char **out) {
    int32_t status = YAR_HOST_OK;
    char *buf;
    size_t i;

    *out = NULL;
    if (value.len < 0 || (value.ptr == NULL && value.len != 0)) {
        return YAR_HOST_INVALID_ARGUMENT;
    }
    if (!allow_empty && value.len == 0) {
        return YAR_HOST_INVALID_ARGUMENT;
    }

    buf = yar_fs_copy_cstr(value.ptr != NULL ? value.ptr : "", (size_t)value.len);
    for (i = 0; i < (size_t)value.len; i++) {
        if (buf[i] == '\0') {
            free(buf);
            return YAR_HOST_INVALID_ARGUMENT;
        }
    }

    *out = buf;
    return status;
}

static void yar_process_free_argv(char **argv, int32_t argc) {
    int32_t i;

    if (argv == NULL) {
        return;
    }
    for (i = 0; i < argc; i++) {
        free(argv[i]);
    }
    free(argv);
}

static int32_t yar_process_argv_cstr(yar_slice argv, char ***out_argv, int32_t *out_argc) {
    yar_str *values;
    char **cargv;
    int32_t i;
    int32_t status;

    *out_argv = NULL;
    *out_argc = 0;
    if (argv.len <= 0 || argv.cap < argv.len || (argv.ptr == NULL && argv.len != 0)) {
        return YAR_HOST_INVALID_ARGUMENT;
    }

    values = (yar_str *)argv.ptr;
    cargv = (char **)calloc((size_t)argv.len + 1, sizeof(char *));
    if (cargv == NULL) {
        yar_trap_oom();
    }

    for (i = 0; i < argv.len; i++) {
        status = yar_host_string_cstr(values[i], i != 0, &cargv[i]);
        if (status != YAR_HOST_OK) {
            yar_process_free_argv(cargv, i);
            return status;
        }
    }

    *out_argv = cargv;
    *out_argc = argv.len;
    return YAR_HOST_OK;
}

void yar_process_args(yar_slice *out) {
    yar_str *values;
    int32_t i;

    if (out == NULL) {
        return;
    }
    *out = yar_empty_slice();
    if (yar_host_argc <= 0 || yar_host_argv == NULL) {
        return;
    }

    values = (yar_str *)yar_alloc_zeroed((long long)((size_t)yar_host_argc * sizeof(yar_str)));
    for (i = 0; i < yar_host_argc; i++) {
        const char *arg = yar_host_argv[i] != NULL ? yar_host_argv[i] : "";
        values[i] = yar_fs_copy_bytes(arg, strlen(arg));
    }
    out->ptr = values;
    out->len = yar_host_argc;
    out->cap = yar_host_argc;
}

int32_t yar_env_lookup(yar_str name, yar_str *out) {
    char *cname;
    const char *value;
    int32_t status;

    if (out == NULL) {
        return YAR_HOST_IO;
    }
    *out = yar_empty_str();

    status = yar_host_string_cstr(name, 1, &cname);
    if (status != YAR_HOST_OK) {
        return status;
    }

    value = getenv(cname);
    free(cname);
    if (value == NULL) {
        return YAR_HOST_NOT_FOUND;
    }

    *out = yar_fs_copy_bytes(value, strlen(value));
    return YAR_HOST_OK;
}

/* --- Windows process helpers --- */

static char *yar_process_build_cmdline(char **cargv, int32_t argc) {
    size_t total = 0;
    int32_t i;
    char *cmdline;
    char *p;

    for (i = 0; i < argc; i++) {
        total += strlen(cargv[i]) * 2 + 3; /* worst-case backslash doubling + quotes + space */
    }
    cmdline = (char *)malloc(total + 1);
    if (cmdline == NULL) {
        yar_trap_oom();
    }
    p = cmdline;
    for (i = 0; i < argc; i++) {
        const char *arg = cargv[i];
        int need_quote = (strchr(arg, ' ') != NULL || strchr(arg, '\t') != NULL || arg[0] == '\0');
        if (i > 0) {
            *p++ = ' ';
        }
        if (need_quote) {
            *p++ = '"';
        }
        while (*arg) {
            int num_backslashes = 0;
            while (*arg == '\\') {
                arg++;
                num_backslashes++;
            }
            if (*arg == '\0') {
                /* backslashes before closing quote: double them */
                int j;
                for (j = 0; j < num_backslashes * 2; j++) {
                    *p++ = '\\';
                }
                break;
            } else if (*arg == '"') {
                /* backslashes before quote: double them + escape quote */
                int j;
                for (j = 0; j < num_backslashes * 2 + 1; j++) {
                    *p++ = '\\';
                }
                *p++ = *arg++;
            } else {
                int j;
                for (j = 0; j < num_backslashes; j++) {
                    *p++ = '\\';
                }
                *p++ = *arg++;
            }
        }
        if (need_quote) {
            *p++ = '"';
        }
    }
    *p = '\0';
    return cmdline;
}

static int32_t yar_process_read_pipe(HANDLE pipe, yar_str *out) {
    char buf[4096];
    char *result = NULL;
    size_t total = 0;
    DWORD bytes_read;

    *out = yar_empty_str();
    for (;;) {
        if (!ReadFile(pipe, buf, sizeof(buf), &bytes_read, NULL) || bytes_read == 0) {
            break;
        }
        result = (char *)realloc(result, total + bytes_read);
        if (result == NULL) {
            yar_trap_oom();
        }
        memcpy(result + total, buf, bytes_read);
        total += bytes_read;
    }
    if (total > 0 && result != NULL) {
        char *managed = (char *)yar_alloc((long long)total);
        memcpy(managed, result, total);
        free(result);
        out->ptr = managed;
        out->len = (long long)total;
    } else {
        free(result);
    }
    return YAR_HOST_OK;
}

int32_t yar_process_run(const yar_slice *argv, yar_process_result *out) {
    char **cargv;
    int32_t argc;
    int32_t status;
    char *cmdline;
    HANDLE stdout_read = NULL, stdout_write = NULL;
    HANDLE stderr_read = NULL, stderr_write = NULL;
    SECURITY_ATTRIBUTES sa;
    STARTUPINFOA si;
    PROCESS_INFORMATION pi;
    DWORD exit_code;

    if (out != NULL) {
        out->exit_code = 0;
        out->stdout = yar_empty_str();
        out->stderr = yar_empty_str();
    }
    if (argv == NULL) {
        return YAR_HOST_INVALID_ARGUMENT;
    }
    status = yar_process_argv_cstr(*argv, &cargv, &argc);
    if (status != YAR_HOST_OK) {
        return status;
    }

    sa.nLength = sizeof(sa);
    sa.bInheritHandle = TRUE;
    sa.lpSecurityDescriptor = NULL;

    if (!CreatePipe(&stdout_read, &stdout_write, &sa, 0) ||
        !CreatePipe(&stderr_read, &stderr_write, &sa, 0)) {
        status = yar_host_status_from_lasterror(GetLastError());
        if (stdout_read) CloseHandle(stdout_read);
        if (stdout_write) CloseHandle(stdout_write);
        if (stderr_read) CloseHandle(stderr_read);
        if (stderr_write) CloseHandle(stderr_write);
        yar_process_free_argv(cargv, argc);
        return status;
    }
    SetHandleInformation(stdout_read, HANDLE_FLAG_INHERIT, 0);
    SetHandleInformation(stderr_read, HANDLE_FLAG_INHERIT, 0);

    memset(&si, 0, sizeof(si));
    si.cb = sizeof(si);
    si.dwFlags = STARTF_USESTDHANDLES;
    si.hStdInput = GetStdHandle(STD_INPUT_HANDLE);
    si.hStdOutput = stdout_write;
    si.hStdError = stderr_write;

    memset(&pi, 0, sizeof(pi));
    cmdline = yar_process_build_cmdline(cargv, argc);

    if (!CreateProcessA(NULL, cmdline, NULL, NULL, TRUE, 0, NULL, NULL, &si, &pi)) {
        status = yar_host_status_from_lasterror(GetLastError());
        free(cmdline);
        CloseHandle(stdout_read);
        CloseHandle(stdout_write);
        CloseHandle(stderr_read);
        CloseHandle(stderr_write);
        yar_process_free_argv(cargv, argc);
        return status;
    }
    free(cmdline);
    yar_process_free_argv(cargv, argc);

    CloseHandle(stdout_write);
    CloseHandle(stderr_write);

    if (out != NULL) {
        yar_process_read_pipe(stdout_read, &out->stdout);
        yar_process_read_pipe(stderr_read, &out->stderr);
    }
    CloseHandle(stdout_read);
    CloseHandle(stderr_read);

    WaitForSingleObject(pi.hProcess, INFINITE);
    GetExitCodeProcess(pi.hProcess, &exit_code);
    if (out != NULL) {
        out->exit_code = (int32_t)exit_code;
    }
    CloseHandle(pi.hProcess);
    CloseHandle(pi.hThread);
    return YAR_HOST_OK;
}

int32_t yar_process_run_inherit(const yar_slice *argv, int32_t *exit_code_out) {
    char **cargv;
    int32_t argc;
    int32_t status;
    char *cmdline;
    STARTUPINFOA si;
    PROCESS_INFORMATION pi;
    DWORD exit_code;

    if (exit_code_out == NULL) {
        return YAR_HOST_IO;
    }
    *exit_code_out = 0;

        return YAR_HOST_INVALID_ARGUMENT;

    *exit_code_out = (int32_t)exit_code;
    CloseHandle(pi.hProcess);
    CloseHandle(pi.hThread);
    return YAR_HOST_OK;
}

static int32_t yar_process_create_tempfile(const char *label, int *fd_out) {
    const char *tmpdir = getenv("TMPDIR");
    size_t dir_len;
    size_t label_len = strlen(label);
    char *pattern;
    int fd;

    if (fd_out == NULL) {
        return YAR_HOST_IO;
    }
    if (tmpdir == NULL || tmpdir[0] == '\0') {
        tmpdir = "/tmp";
    }

    dir_len = strlen(tmpdir);
    pattern = (char *)malloc(dir_len + label_len + sizeof("/yar--XXXXXX"));
    if (pattern == NULL) {
        yar_trap_oom();
    }
    memcpy(pattern, tmpdir, dir_len);
    pattern[dir_len] = '/';
    memcpy(pattern + dir_len + 1, "yar-", 4);
    memcpy(pattern + dir_len + 5, label, label_len);
    memcpy(pattern + dir_len + 5 + label_len, "-XXXXXX", 8);

    fd = mkstemp(pattern);
    if (fd < 0) {
        int32_t status = yar_host_status_from_errno(errno);
        free(pattern);
        return status;
    }
    (void)unlink(pattern);
    free(pattern);
    *fd_out = fd;
    return YAR_HOST_OK;
}

static int32_t yar_process_read_exec_error(int fd, int *errnum_out) {
    int errnum = 0;
    size_t off = 0;

    *errnum_out = 0;
    while (off < sizeof(errnum)) {
        ssize_t n = read(fd, ((char *)&errnum) + off, sizeof(errnum) - off);
        if (n == 0) {
            if (off == 0) {
                return YAR_HOST_OK;
            }
            return YAR_HOST_IO;
        }
        if (n < 0) {
            if (errno == EINTR) {
                continue;
            }
            return yar_host_status_from_errno(errno);
        }
        off += (size_t)n;
    }
    *errnum_out = errnum;
    return YAR_HOST_OK;
}

static int32_t yar_process_wait(pid_t pid, int32_t *exit_code_out) {
    int status = 0;
    pid_t waited;

    do {
        waited = waitpid(pid, &status, 0);
    } while (waited < 0 && errno == EINTR);
    if (waited < 0) {
        return yar_host_status_from_errno(errno);
    }

    if (WIFEXITED(status)) {
        *exit_code_out = (int32_t)WEXITSTATUS(status);
        return YAR_HOST_OK;
    }
    if (WIFSIGNALED(status)) {
        *exit_code_out = 128 + (int32_t)WTERMSIG(status);
        return YAR_HOST_OK;
    }
    return YAR_HOST_IO;
}

static int32_t yar_process_read_file_fd(int fd, yar_str *out) {
    struct stat st;
    char *buf;
    size_t total;
    size_t off;

    *out = yar_empty_str();
    if (fstat(fd, &st) != 0) {
        return yar_host_status_from_errno(errno);
    }
    if (st.st_size == 0) {
        return YAR_HOST_OK;
    }
    if (st.st_size < 0) {
        return YAR_HOST_IO;
    }
    if (lseek(fd, 0, SEEK_SET) < 0) {
        return yar_host_status_from_errno(errno);
    }

    total = (size_t)st.st_size;
    buf = (char *)yar_alloc((long long)total);

    off = 0;
    while (off < total) {
        ssize_t n = read(fd, buf + off, total - off);
        if (n < 0) {
            if (errno == EINTR) {
                continue;
            }
            return yar_host_status_from_errno(errno);
        }
        if (n == 0) {
            break;
        }
        off += (size_t)n;
    }

    out->ptr = buf;
    out->len = (long long)off;
    return YAR_HOST_OK;
}

static int32_t yar_process_collect_capture(int stdout_fd, int stderr_fd, int32_t exit_code, yar_process_result *out) {
    int32_t status;
    yar_str stdout_text;
    yar_str stderr_text;

    if (out == NULL) {
        return YAR_HOST_IO;
    }
    out->exit_code = exit_code;
    out->stdout = yar_empty_str();
    out->stderr = yar_empty_str();

    status = yar_process_read_file_fd(stdout_fd, &stdout_text);
    if (status != YAR_HOST_OK) {
        return status;
    }
    status = yar_process_read_file_fd(stderr_fd, &stderr_text);
    if (status != YAR_HOST_OK) {
        yar_free_str(stdout_text);
        return status;
    }

    out->stdout = stdout_text;
    out->stderr = stderr_text;
    return YAR_HOST_OK;
}

int32_t yar_process_run(const yar_slice *argv, yar_process_result *out) {
    char **cargv;
    int32_t argc;
    int stdout_fd = -1;
    int stderr_fd = -1;
    int exec_pipe[2] = {-1, -1};
    int exec_errnum;
    int32_t exit_code;
    int32_t status;
    pid_t pid;

    if (out != NULL) {
        out->exit_code = 0;
        out->stdout = yar_empty_str();
        out->stderr = yar_empty_str();
    }

    if (argv == NULL) {
        return YAR_HOST_INVALID_ARGUMENT;
    }
    status = yar_process_argv_cstr(*argv, &cargv, &argc);
    if (status != YAR_HOST_OK) {
        return status;
    }
    status = yar_process_create_tempfile("stdout", &stdout_fd);
    if (status != YAR_HOST_OK) {
        yar_process_free_argv(cargv, argc);
        return status;
    }
    status = yar_process_create_tempfile("stderr", &stderr_fd);
    if (status != YAR_HOST_OK) {
        close(stdout_fd);
        yar_process_free_argv(cargv, argc);
        return status;
    }
    if (pipe(exec_pipe) != 0) {
        status = yar_host_status_from_errno(errno);
        close(stdout_fd);
        close(stderr_fd);
        yar_process_free_argv(cargv, argc);
        return status;
    }
    if (fcntl(exec_pipe[1], F_SETFD, FD_CLOEXEC) < 0) {
        status = yar_host_status_from_errno(errno);
        close(exec_pipe[0]);
        close(exec_pipe[1]);
        close(stdout_fd);
        close(stderr_fd);
        yar_process_free_argv(cargv, argc);
        return status;
    }

    pid = fork();
    if (pid < 0) {
        status = yar_host_status_from_errno(errno);
        close(exec_pipe[0]);
        close(exec_pipe[1]);
        close(stdout_fd);
        close(stderr_fd);
        yar_process_free_argv(cargv, argc);
        return status;
    }
    if (pid == 0) {
        int child_errno;

        close(exec_pipe[0]);
        if (dup2(stdout_fd, STDOUT_FILENO) < 0 || dup2(stderr_fd, STDERR_FILENO) < 0) {
            child_errno = errno;
            (void)write(exec_pipe[1], &child_errno, sizeof(child_errno));
            _exit(127);
        }
        close(stdout_fd);
        close(stderr_fd);
        execvp(cargv[0], cargv);
        child_errno = errno;
        (void)write(exec_pipe[1], &child_errno, sizeof(child_errno));
        _exit(127);
    }

    close(exec_pipe[1]);
    status = yar_process_read_exec_error(exec_pipe[0], &exec_errnum);
    close(exec_pipe[0]);
    if (status != YAR_HOST_OK) {
        close(stdout_fd);
        close(stderr_fd);
        yar_process_free_argv(cargv, argc);
        return status;
    }
    if (exec_errnum != 0) {
        close(stdout_fd);
        close(stderr_fd);
        (void)yar_process_wait(pid, &exit_code);
        yar_process_free_argv(cargv, argc);
        return yar_host_status_from_errno(exec_errnum);
    }

    status = yar_process_wait(pid, &exit_code);
    if (status == YAR_HOST_OK) {
        status = yar_process_collect_capture(stdout_fd, stderr_fd, exit_code, out);
    }

    close(stdout_fd);
    close(stderr_fd);
    yar_process_free_argv(cargv, argc);
    return status;
}

int32_t yar_process_run_inherit(const yar_slice *argv, int32_t *exit_code_out) {
    char **cargv;
    int32_t argc;
    int exec_pipe[2] = {-1, -1};
    int exec_errnum;
    int32_t exit_code = 0;
    int32_t status;
    pid_t pid;

    if (exit_code_out == NULL) {
        return YAR_HOST_IO;
    }
    *exit_code_out = 0;

    if (argv == NULL) {
        return YAR_HOST_INVALID_ARGUMENT;
    }
    status = yar_process_argv_cstr(*argv, &cargv, &argc);
    if (status != YAR_HOST_OK) {
        return status;
    }
    if (pipe(exec_pipe) != 0) {
        status = yar_host_status_from_errno(errno);
        yar_process_free_argv(cargv, argc);
        return status;
    }
    if (fcntl(exec_pipe[1], F_SETFD, FD_CLOEXEC) < 0) {
        status = yar_host_status_from_errno(errno);
        close(exec_pipe[0]);
        close(exec_pipe[1]);
        yar_process_free_argv(cargv, argc);
        return status;
    }

    pid = fork();
    if (pid < 0) {

        close(exec_pipe[0]);
        close(exec_pipe[1]);
        yar_process_free_argv(cargv, argc);
        return status;
    }
    if (pid == 0) {
        int child_errno;

        close(exec_pipe[0]);
        execvp(cargv[0], cargv);

        (void)write(exec_pipe[1], &child_errno, sizeof(child_errno));
        _exit(127);
    }

    if (status == YAR_HOST_OK) {
        *exit_code_out = exit_code;
    }
    yar_process_free_argv(cargv, argc);
    return status;
}

/* --- filesystem runtime --- */

#define YAR_FS_OK                0
#define YAR_FS_NOT_FOUND         1
#define YAR_FS_PERMISSION_DENIED 2
#define YAR_FS_ALREADY_EXISTS    3
#define YAR_FS_INVALID_PATH      4
#define YAR_FS_IO                5

#define YAR_FS_KIND_FILE      0
#define YAR_FS_KIND_DIRECTORY 1
#define YAR_FS_KIND_OTHER     2

typedef struct {
    yar_str  name;
    uint8_t  is_dir;
} yar_fs_dir_entry;

static yar_str yar_empty_str(void) {
    yar_str result;
    result.ptr = NULL;
    result.len = 0;
    return result;
}

static yar_slice yar_empty_slice(void) {
    yar_slice result;
    result.ptr = NULL;
    result.len = 0;
    result.cap = 0;
    return result;
}

#ifdef _WIN32
static int32_t yar_fs_status_from_lasterror(DWORD err) {
    switch (err) {
    case ERROR_FILE_NOT_FOUND:
    case ERROR_PATH_NOT_FOUND:
        return YAR_FS_NOT_FOUND;
    case ERROR_ACCESS_DENIED:
        return YAR_FS_PERMISSION_DENIED;
    case ERROR_ALREADY_EXISTS:
    case ERROR_FILE_EXISTS:
        return YAR_FS_ALREADY_EXISTS;
    case ERROR_INVALID_NAME:
    case ERROR_BAD_PATHNAME:
        return YAR_FS_INVALID_PATH;
    default:
        return YAR_FS_IO;
    }
}

static int32_t yar_fs_status_from_errno(int errnum) {
    switch (errnum) {
    case 0:
        return YAR_FS_IO;
    case ENOENT:
        return YAR_FS_NOT_FOUND;
    case EACCES:
        return YAR_FS_PERMISSION_DENIED;
    case EEXIST:
        return YAR_FS_ALREADY_EXISTS;
    default:
        return YAR_FS_IO;
    }
}
#else
static int32_t yar_fs_status_from_errno(int errnum) {
    switch (errnum) {
    case 0:
        return YAR_FS_IO;
    case ENOENT:
    case ENOTDIR:
        return YAR_FS_NOT_FOUND;
    case EACCES:
    case EPERM:
        return YAR_FS_PERMISSION_DENIED;
    case EEXIST:
        return YAR_FS_ALREADY_EXISTS;
    case EINVAL:
    case ENAMETOOLONG:
    case ELOOP:
        return YAR_FS_INVALID_PATH;
    default:
        return YAR_FS_IO;
    }
}
#endif

static char *yar_fs_copy_cstr(const char *src, size_t len) {
    char *dst = (char *)malloc(len + 1);
    if (dst == NULL) {
        yar_trap_oom();
    }
    if (len > 0) {
        memcpy(dst, src, len);
    }
    dst[len] = '\0';
    return dst;
}

static yar_str yar_fs_copy_bytes(const char *src, size_t len) {
    yar_str result;
    if (len == 0) {
        return yar_empty_str();
    }
    result.ptr = (const char *)yar_alloc((long long)len);
    memcpy((void *)result.ptr, src, len);
    result.len = (long long)len;
    return result;
}

static char *yar_fs_path_cstr(yar_str path, int32_t *status_out) {
    size_t i;
    char *buf;

    if (path.len < 0 || (path.ptr == NULL && path.len != 0)) {
        *status_out = YAR_FS_INVALID_PATH;
        return NULL;
    }

    buf = yar_fs_copy_cstr(path.ptr != NULL ? path.ptr : "", (size_t)path.len);
    for (i = 0; i < (size_t)path.len; i++) {
        if (buf[i] == '\0') {
            free(buf);
            *status_out = YAR_FS_INVALID_PATH;
            return NULL;
        }
    }

    *status_out = YAR_FS_OK;
    return buf;
}

static char *yar_fs_join_path(const char *dir, const char *name) {
    size_t dir_len = strlen(dir);
    size_t name_len = strlen(name);
#ifdef _WIN32
    char sep = '\\';
    int needs_sep = dir_len > 0 && dir[dir_len - 1] != '\\' && dir[dir_len - 1] != '/';
#else
    char sep = '/';
    int needs_sep = dir_len > 0 && dir[dir_len - 1] != '/';
#endif
    char *path = (char *)malloc(dir_len + (size_t)needs_sep + name_len + 1);

    if (path == NULL) {
        yar_trap_oom();
    }
    memcpy(path, dir, dir_len);
    if (needs_sep) {
        path[dir_len] = sep;
    }
    memcpy(path + dir_len + (size_t)needs_sep, name, name_len);
    path[dir_len + (size_t)needs_sep + name_len] = '\0';
    return path;
}

static void yar_fs_free_dir_entries(yar_fs_dir_entry *entries, int32_t count) {
    (void)count;
    if (entries == NULL) {
        return;
    }
    free(entries);
}

int32_t yar_fs_write_file(yar_str path, yar_str data) {
    int32_t status;
    char *cpath = yar_fs_path_cstr(path, &status);
    FILE *file;

    if (cpath == NULL) {
        return status;
    }

    file = fopen(cpath, "wb");
    if (file == NULL) {
        status = yar_fs_status_from_errno(errno);
        free(cpath);
        return status;
    }
    if (data.len > 0) {
        if (fwrite(data.ptr, 1, (size_t)data.len, file) != (size_t)data.len) {
            status = ferror(file) ? yar_fs_status_from_errno(errno) : YAR_FS_IO;
            (void)fclose(file);
            free(cpath);
            return status;
        }
    }
    if (fclose(file) != 0) {
        free(cpath);
        return yar_fs_status_from_errno(errno);
    }
    free(cpath);
    return YAR_FS_OK;
}

int32_t yar_fs_read_file(yar_str path, yar_str *out) {
    int32_t status;
    char *cpath = yar_fs_path_cstr(path, &status);
    FILE *file;
    long length;
    char *buf;

    *out = yar_empty_str();
    if (cpath == NULL) {
        return status;
    }

    file = fopen(cpath, "rb");
    if (file == NULL) {
        status = yar_fs_status_from_errno(errno);
        free(cpath);
        return status;
    }
    if (fseek(file, 0, SEEK_END) != 0) {
        status = YAR_FS_IO;
        (void)fclose(file);
        free(cpath);
        return status;
    }
    length = ftell(file);
    if (length < 0) {
        status = YAR_FS_IO;
        (void)fclose(file);
        free(cpath);
        return status;
    }
    if (fseek(file, 0, SEEK_SET) != 0) {
        status = YAR_FS_IO;
        (void)fclose(file);
        free(cpath);
        return status;
    }

    if (length == 0) {
        if (fclose(file) != 0) {
            status = yar_fs_status_from_errno(errno);
        }
        free(cpath);
        return status;
    }

    buf = (char *)yar_alloc(length);
    if (fread(buf, 1, (size_t)length, file) != (size_t)length) {
        status = ferror(file) ? yar_fs_status_from_errno(errno) : YAR_FS_IO;
        (void)fclose(file);
        free(cpath);
        return status;
    }
    if (fclose(file) != 0) {
        status = yar_fs_status_from_errno(errno);
        free(cpath);
        return status;
    }

    out->ptr = buf;
    out->len = (long long)length;
    free(cpath);
    return YAR_FS_OK;
}

#ifdef _WIN32
int32_t yar_fs_read_dir(yar_str path, yar_slice *out) {
    int32_t status;
    char *cpath = yar_fs_path_cstr(path, &status);
    char *pattern;
    size_t cpath_len;
    WIN32_FIND_DATAA fd;
    HANDLE hFind;
    yar_fs_dir_entry *entries = NULL;
    int32_t count = 0;
    int32_t cap = 0;

    *out = yar_empty_slice();
    if (cpath == NULL) {
        return status;
    }

    cpath_len = strlen(cpath);
    pattern = (char *)malloc(cpath_len + 3);
    if (pattern == NULL) {
        free(cpath);
        yar_trap_oom();
    }
    memcpy(pattern, cpath, cpath_len);
    pattern[cpath_len] = '\\';
    pattern[cpath_len + 1] = '*';
    pattern[cpath_len + 2] = '\0';

    hFind = FindFirstFileA(pattern, &fd);
    free(pattern);
    if (hFind == INVALID_HANDLE_VALUE) {
        status = yar_fs_status_from_lasterror(GetLastError());
        free(cpath);
        return status;
    }

    do {
        if (strcmp(fd.cFileName, ".") == 0 || strcmp(fd.cFileName, "..") == 0) {
            continue;
        }
        if (count == cap) {
            int32_t next_cap = cap == 0 ? 8 : cap * 2;
            yar_fs_dir_entry *next_entries = (yar_fs_dir_entry *)realloc(entries, (size_t)next_cap * sizeof(yar_fs_dir_entry));
            if (next_entries == NULL) {
                yar_fs_free_dir_entries(entries, count);
                FindClose(hFind);
                free(cpath);
                yar_trap_oom();
            }
            entries = next_entries;
            cap = next_cap;
        }
        entries[count].name = yar_fs_copy_bytes(fd.cFileName, strlen(fd.cFileName));
        entries[count].is_dir = (fd.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) ? 1 : 0;
        count++;
    } while (FindNextFileA(hFind, &fd));

    FindClose(hFind);
    if (count == 0) {
        free(entries);
        entries = NULL;
    } else {
        size_t final_size = (size_t)count * sizeof(yar_fs_dir_entry);
        yar_fs_dir_entry *final_entries = (yar_fs_dir_entry *)yar_alloc((long long)final_size);
        memcpy(final_entries, entries, final_size);
        free(entries);
        entries = final_entries;
    }
    out->ptr = entries;
    out->len = count;
    out->cap = count;
    free(cpath);
    return YAR_FS_OK;
}
#else
int32_t yar_fs_read_dir(yar_str path, yar_slice *out) {
    int32_t status;
    char *cpath = yar_fs_path_cstr(path, &status);
    DIR *dir;
    struct dirent *entry;
    yar_fs_dir_entry *entries = NULL;
    int32_t count = 0;
    int32_t cap = 0;

    *out = yar_empty_slice();
    if (cpath == NULL) {
        return status;
    }

    dir = opendir(cpath);
    if (dir == NULL) {
        status = yar_fs_status_from_errno(errno);
        free(cpath);
        return status;
    }

    while ((entry = readdir(dir)) != NULL) {
        char *child_path;
        struct stat st;

        if ((strcmp(entry->d_name, ".") == 0) || (strcmp(entry->d_name, "..") == 0)) {
            continue;
        }
        if (count == cap) {
            int32_t next_cap = cap == 0 ? 8 : cap * 2;
            yar_fs_dir_entry *next_entries = (yar_fs_dir_entry *)realloc(entries, (size_t)next_cap * sizeof(yar_fs_dir_entry));
            if (next_entries == NULL) {
                yar_fs_free_dir_entries(entries, count);
                closedir(dir);
                free(cpath);
                yar_trap_oom();
            }
            entries = next_entries;
            cap = next_cap;
        }

        child_path = yar_fs_join_path(cpath, entry->d_name);
        if (stat(child_path, &st) != 0) {
            free(child_path);
            yar_fs_free_dir_entries(entries, count);
            (void)closedir(dir);
            free(cpath);
            return yar_fs_status_from_errno(errno);
        }
        free(child_path);

        entries[count].name = yar_fs_copy_bytes(entry->d_name, strlen(entry->d_name));
        entries[count].is_dir = S_ISDIR(st.st_mode) ? 1 : 0;
        count++;
    }

    if (closedir(dir) != 0) {
        yar_fs_free_dir_entries(entries, count);
        free(cpath);
        return yar_fs_status_from_errno(errno);
    }

    if (count == 0) {
        free(entries);
        entries = NULL;
    } else {
        size_t final_size = (size_t)count * sizeof(yar_fs_dir_entry);
        yar_fs_dir_entry *final_entries = (yar_fs_dir_entry *)yar_alloc((long long)final_size);
        memcpy(final_entries, entries, final_size);
        free(entries);
        entries = final_entries;
    }

    out->ptr = entries;
    out->len = count;
    out->cap = count;
    free(cpath);
    return YAR_FS_OK;
}
#endif

static int32_t yar_fs_mkdir_one(const char *path) {

        return YAR_FS_INVALID_PATH;

    attrs = GetFileAttributesA(path);
    if (attrs != INVALID_FILE_ATTRIBUTES) {
        return (attrs & FILE_ATTRIBUTE_DIRECTORY) ? YAR_FS_OK : YAR_FS_ALREADY_EXISTS;
    }
    if (!CreateDirectoryA(path, NULL)) {
        return yar_fs_status_from_lasterror(GetLastError());
    }
    return YAR_FS_OK;

    if (path[0] == '\0') {
        return YAR_FS_INVALID_PATH;

    if (stat(path, &st) == 0) {
        return S_ISDIR(st.st_mode) ? YAR_FS_OK : YAR_FS_ALREADY_EXISTS;
    }
    if (errno != ENOENT) {
        return yar_fs_status_from_errno(errno);

        return yar_fs_status_from_errno(errno);

int32_t yar_fs_mkdir_all(yar_str path) {
    int32_t status;
    char *cpath = yar_fs_path_cstr(path, &status);
    size_t i;
    size_t start = 0;
    size_t len;

    if (cpath == NULL) {
        return status;
    }

    len = strlen(cpath);
    if (len == 0) {
        free(cpath);
        return YAR_FS_INVALID_PATH;
    }
    while (len > 1 && cpath[len - 1] == '/') {
        cpath[len - 1] = '\0';
        len--;
    }
    if (cpath[0] == '/') {
        start = 1;
    } else if (len >= 3 && cpath[1] == ':' && cpath[2] == '/') {
        start = 3;
    }

    for (i = start; cpath[i] != '\0'; i++) {
        if (cpath[i] == '/') {
            char saved = cpath[i];
            cpath[i] = '\0';
            status = yar_fs_mkdir_one(cpath);
            cpath[i] = saved;
            if (status != YAR_FS_OK) {
                free(cpath);
                return status;
            }
        }
    }

    status = yar_fs_mkdir_one(cpath);
    free(cpath);
    return status;
}

#ifdef _WIN32
static int32_t yar_fs_remove_all_c(const char *path) {
    DWORD attrs = GetFileAttributesA(path);
    if (attrs == INVALID_FILE_ATTRIBUTES) {
        DWORD err = GetLastError();
        return (err == ERROR_FILE_NOT_FOUND || err == ERROR_PATH_NOT_FOUND)
            ? YAR_FS_OK : yar_fs_status_from_lasterror(err);
    }
    if (attrs & FILE_ATTRIBUTE_DIRECTORY) {
        char *pattern;
        size_t path_len = strlen(path);
        WIN32_FIND_DATAA fd;
        HANDLE hFind;

        pattern = (char *)malloc(path_len + 3);
        if (pattern == NULL) {
            yar_trap_oom();
        }
        memcpy(pattern, path, path_len);
        pattern[path_len] = '\\';
        pattern[path_len + 1] = '*';
        pattern[path_len + 2] = '\0';

        hFind = FindFirstFileA(pattern, &fd);
        free(pattern);
        if (hFind != INVALID_HANDLE_VALUE) {
            do {
                char *child;
                int32_t status;
                if (strcmp(fd.cFileName, ".") == 0 || strcmp(fd.cFileName, "..") == 0) {
                    continue;
                }
                child = yar_fs_join_path(path, fd.cFileName);
                status = yar_fs_remove_all_c(child);
                free(child);
                if (status != YAR_FS_OK) {
                    FindClose(hFind);
                    return status;
                }
            } while (FindNextFileA(hFind, &fd));
            FindClose(hFind);
        }
        if (!RemoveDirectoryA(path)) {
            return yar_fs_status_from_lasterror(GetLastError());
        }
        return YAR_FS_OK;
    }
    if (!DeleteFileA(path)) {
        return yar_fs_status_from_lasterror(GetLastError());
    }
    return YAR_FS_OK;
}
#else
static int32_t yar_fs_remove_all_c(const char *path) {
    struct stat st;

    if (stat(path, &st) != 0) {
        return errno == ENOENT ? YAR_FS_OK : yar_fs_status_from_errno(errno);
    }
    if (S_ISDIR(st.st_mode)) {
        DIR *dir = opendir(path);
        struct dirent *entry;
        if (dir == NULL) {
            return yar_fs_status_from_errno(errno);
        }
        while ((entry = readdir(dir)) != NULL) {
            char *child;
            int32_t status;

            if ((strcmp(entry->d_name, ".") == 0) || (strcmp(entry->d_name, "..") == 0)) {
                continue;
            }
            child = yar_fs_join_path(path, entry->d_name);
            status = yar_fs_remove_all_c(child);
            free(child);
            if (status != YAR_FS_OK) {
                (void)closedir(dir);
                return status;
            }
        }
        if (closedir(dir) != 0) {
            return yar_fs_status_from_errno(errno);
        }
        if (rmdir(path) != 0) {
            return yar_fs_status_from_errno(errno);
        }
        return YAR_FS_OK;
    }
    if (remove(path) != 0) {
        return yar_fs_status_from_errno(errno);
    }
    return YAR_FS_OK;
}

    if (cpath == NULL) {

int32_t yar_fs_temp_dir(yar_str prefix, yar_str *out) {
    char tmp_path[MAX_PATH];
    char tmp_file[MAX_PATH];
    char *dir_path;
    size_t prefix_len;
    size_t i;
    char prefix_buf[64];

    *out = yar_empty_str();
    if (prefix.len < 0 || (prefix.ptr == NULL && prefix.len != 0)) {
        return YAR_FS_INVALID_PATH;
    }
    for (i = 0; i < (size_t)prefix.len; i++) {
        if (prefix.ptr[i] == '\0' || prefix.ptr[i] == '/' || prefix.ptr[i] == '\\') {
            return YAR_FS_INVALID_PATH;
        }
    }

    if (GetTempPathA(MAX_PATH, tmp_path) == 0) {
        return yar_fs_status_from_lasterror(GetLastError());
    }

    prefix_len = (size_t)prefix.len;
    if (prefix_len > 3) {
        prefix_len = 3; /* GetTempFileNameA uses only first 3 chars of prefix */
    }
    memset(prefix_buf, 0, sizeof(prefix_buf));
    if (prefix_len > 0) {
        memcpy(prefix_buf, prefix.ptr, prefix_len);
    }

    if (GetTempFileNameA(tmp_path, prefix_buf, 0, tmp_file) == 0) {
        return yar_fs_status_from_lasterror(GetLastError());
    }
    /* GetTempFileNameA creates a file; delete it and create a directory instead */
    DeleteFileA(tmp_file);
    if (!CreateDirectoryA(tmp_file, NULL)) {
        return yar_fs_status_from_lasterror(GetLastError());
    }

    dir_path = yar_fs_copy_cstr(tmp_file, strlen(tmp_file));
    out->ptr = dir_path;
    out->len = (long long)strlen(dir_path);
    return YAR_FS_OK;
}

int32_t yar_fs_temp_dir(yar_str prefix, yar_str *out) {
    const char *base;
    size_t base_len;
    size_t prefix_len;
    int needs_sep;
    char *tmpl;
    size_t i;
    int fd;

    *out = yar_empty_str();

        return YAR_FS_INVALID_PATH;
    }

            return YAR_FS_INVALID_PATH;

        base = "/tmp";
    }
    base_len = strlen(base);

    needs_sep = base_len > 0 && base[base_len - 1] != '/';

    if (tmpl == NULL) {

        free(tmpl);

    return YAR_FS_OK;
}

/* --- map runtime --- */

#define YAR_MAP_INIT_CAP 8
#define YAR_MAP_LOAD_NUM 3
#define YAR_MAP_LOAD_DEN 4

/* key kinds passed from codegen */
#define YAR_KEY_BOOL  0
#define YAR_KEY_I32   1
#define YAR_KEY_I64   2
#define YAR_KEY_STR   3

typedef struct {
    unsigned char *entries;  /* bucket array: each bucket is (1 byte used + key_size + value_size) */
    int32_t        count;
    int32_t        cap;
    int32_t        key_size;
    int32_t        value_size;
    int32_t        bucket_size;
    int32_t        key_kind;
} yar_map;

static uint64_t yar_map_hash(const void *key, int32_t key_kind, int32_t key_size) {
    uint64_t h = 14695981039346656037ULL;
    if (key_kind == YAR_KEY_STR) {
        const yar_str *s = (const yar_str *)key;
        const unsigned char *data = (const unsigned char *)s->ptr;
        long long len = s->len;
        for (long long i = 0; i < len; i++) {
            h ^= data[i];
            h *= 1099511628211ULL;
        }
    } else {
        const unsigned char *data = (const unsigned char *)key;
        for (int32_t i = 0; i < key_size; i++) {
            h ^= data[i];
            h *= 1099511628211ULL;
        }
    }
    return h;
}

static int yar_map_keys_equal(const void *a, const void *b, int32_t key_kind, int32_t key_size) {
    if (key_kind == YAR_KEY_STR) {
        const yar_str *sa = (const yar_str *)a;
        const yar_str *sb = (const yar_str *)b;
        if (sa->len != sb->len) return 0;
        if (sa->len == 0) return 1;
        for (long long i = 0; i < sa->len; i++) {
            if (sa->ptr[i] != sb->ptr[i]) return 0;
        }
        return 1;
    }
    const unsigned char *pa = (const unsigned char *)a;
    const unsigned char *pb = (const unsigned char *)b;
    for (int32_t i = 0; i < key_size; i++) {
        if (pa[i] != pb[i]) return 0;
    }
    return 1;
}

void *yar_map_new(int32_t key_kind, int32_t key_size, int32_t value_size) {
    yar_map *m = (yar_map *)yar_alloc_zeroed((long long)sizeof(yar_map));
    m->count = 0;
    m->cap = YAR_MAP_INIT_CAP;
    m->key_size = key_size;
    m->value_size = value_size;
    m->bucket_size = 1 + key_size + value_size;
    m->key_kind = key_kind;
    m->entries = (unsigned char *)yar_alloc_zeroed((long long)((size_t)m->cap * (size_t)m->bucket_size));
    return m;
}

static void yar_map_grow(yar_map *m);

static unsigned char *yar_map_find_slot(yar_map *m, const void *key, uint64_t hash, int *found) {
    int32_t mask = m->cap - 1;
    int32_t idx = (int32_t)(hash & (uint64_t)mask);
    for (;;) {
        unsigned char *bucket = m->entries + (int64_t)idx * m->bucket_size;
        if (bucket[0] == 0) {
            *found = 0;
            return bucket;
        }
        if (yar_map_keys_equal(bucket + 1, key, m->key_kind, m->key_size)) {
            *found = 1;
            return bucket;
        }
        idx = (idx + 1) & mask;
    }
}

void yar_map_set(void *map_ptr, const void *key, const void *value) {
    yar_map *m = (yar_map *)map_ptr;
    if ((int64_t)(m->count + 1) * YAR_MAP_LOAD_DEN > (int64_t)m->cap * YAR_MAP_LOAD_NUM) {
        yar_map_grow(m);
    }
    uint64_t hash = yar_map_hash(key, m->key_kind, m->key_size);
    int found;
    unsigned char *bucket = yar_map_find_slot(m, key, hash, &found);
    if (!found) {
        bucket[0] = 1;
        for (int32_t i = 0; i < m->key_size; i++) bucket[1 + i] = ((const unsigned char *)key)[i];
        m->count++;
    }
    unsigned char *val_dst = bucket + 1 + m->key_size;
    const unsigned char *val_src = (const unsigned char *)value;
    for (int32_t i = 0; i < m->value_size; i++) val_dst[i] = val_src[i];
}

int32_t yar_map_get(void *map_ptr, const void *key, void *value_out) {
    yar_map *m = (yar_map *)map_ptr;
    uint64_t hash = yar_map_hash(key, m->key_kind, m->key_size);
    int found;
    unsigned char *bucket = yar_map_find_slot(m, key, hash, &found);
    if (!found) return 0;
    unsigned char *val_src = bucket + 1 + m->key_size;
    unsigned char *val_dst = (unsigned char *)value_out;
    for (int32_t i = 0; i < m->value_size; i++) val_dst[i] = val_src[i];
    return 1;
}

int32_t yar_map_has(void *map_ptr, const void *key) {
    yar_map *m = (yar_map *)map_ptr;
    uint64_t hash = yar_map_hash(key, m->key_kind, m->key_size);
    int found;
    yar_map_find_slot(m, key, hash, &found);
    return found ? 1 : 0;
}

void yar_map_delete(void *map_ptr, const void *key) {
    yar_map *m = (yar_map *)map_ptr;
    uint64_t hash = yar_map_hash(key, m->key_kind, m->key_size);
    int found;
    unsigned char *bucket = yar_map_find_slot(m, key, hash, &found);
    if (!found) return;
    /* mark deleted and rehash forward entries */
    bucket[0] = 0;
    m->count--;
    int32_t mask = m->cap - 1;
    int32_t idx = (int32_t)((hash & (uint64_t)mask) + 1) & mask;
    for (;;) {
        unsigned char *next = m->entries + (int64_t)idx * m->bucket_size;
        if (next[0] == 0) break;
        /* remove and reinsert */
        unsigned char tmp[256];
        unsigned char *key_tmp = (m->bucket_size <= 256) ? tmp : (unsigned char *)malloc((size_t)m->bucket_size);
        if (!key_tmp) { yar_trap_oom(); }
        for (int32_t i = 0; i < m->bucket_size; i++) key_tmp[i] = next[i];
        next[0] = 0;
        m->count--;
        /* reinsert */
        uint64_t rh = yar_map_hash(key_tmp + 1, m->key_kind, m->key_size);
        int rf;
        unsigned char *dst = yar_map_find_slot(m, key_tmp + 1, rh, &rf);
        for (int32_t i = 0; i < m->bucket_size; i++) dst[i] = key_tmp[i];
        m->count++;
        if (key_tmp != tmp) free(key_tmp);
        idx = (idx + 1) & mask;
    }
}

int32_t yar_map_len(void *map_ptr) {
    yar_map *m = (yar_map *)map_ptr;
    return m->count;
}

yar_slice yar_map_keys(void *map_ptr) {
    yar_map *m = (yar_map *)map_ptr;
    yar_slice out;
    unsigned char *dst;
    int32_t copied;
    int32_t i;

    if (m->count == 0) {
        return yar_empty_slice();
    }

    dst = (unsigned char *)yar_alloc((long long)((size_t)m->count * (size_t)m->key_size));

    copied = 0;
    for (i = 0; i < m->cap; i++) {
        unsigned char *bucket = m->entries + (int64_t)i * m->bucket_size;
        if (bucket[0] == 0) {
            continue;
        }
        memcpy(dst + (int64_t)copied * m->key_size, bucket + 1, (size_t)m->key_size);
        copied++;
    }

    out.ptr = dst;
    out.len = copied;
    out.cap = copied;
    return out;
}

static void yar_map_grow(yar_map *m) {
    int32_t old_cap = m->cap;
    unsigned char *old_entries = m->entries;
    int32_t new_cap = old_cap * 2;
    m->cap = new_cap;
    m->entries = (unsigned char *)yar_alloc_zeroed((long long)((size_t)new_cap * (size_t)m->bucket_size));
    m->count = 0;
    for (int32_t i = 0; i < old_cap; i++) {
        unsigned char *bucket = old_entries + (int64_t)i * m->bucket_size;
        if (bucket[0] == 0) continue;
        yar_map_set(m, bucket + 1, bucket + 1 + m->key_size);
    }
}

void *yar_alloc_zeroed(long long size) {
    if (size < 0) {
        static const char message[] = "runtime failure: invalid allocation size\n";
        yar_runtime_fail(message, sizeof(message) - 1);
    }
    return yar_gc_alloc_bytes((size_t)size, 1);
}

typedef struct {
    char     *buf;
    long long len;
    long long cap;
} yar_sb;

void *yar_sb_new(void) {
    yar_sb *sb = (yar_sb *)yar_alloc_zeroed((long long)sizeof(yar_sb));
    sb->cap = 64;
    sb->buf = (char *)yar_alloc(sb->cap);
    sb->len = 0;
    return sb;
}

void yar_sb_write(void *handle, const char *data, long long data_len) {
    if (data_len <= 0) return;
    yar_sb *sb = (yar_sb *)handle;
    long long needed = sb->len + data_len;
    if (needed > sb->cap) {
        long long new_cap = sb->cap * 2;
        if (new_cap < needed) new_cap = needed;
        char *new_buf = (char *)yar_alloc(new_cap);
        memcpy(new_buf, sb->buf, (size_t)sb->len);
        sb->buf = new_buf;
        sb->cap = new_cap;
    }
    memcpy(sb->buf + sb->len, data, (size_t)data_len);
    sb->len = needed;
}

yar_str yar_sb_string(void *handle) {
    yar_sb *sb = (yar_sb *)handle;
    yar_str result;
    if (sb->len == 0) {
        return yar_empty_str();
    }
    char *buf = (char *)yar_alloc(sb->len);
    memcpy(buf, sb->buf, (size_t)sb->len);
    result.ptr = buf;
    result.len = sb->len;
    sb->len = 0;
    return result;
}

static void yar_net_ensure_init(void) {
    static int initialized = 0;
    if (initialized) return;
    initialized = 1;
#ifdef _WIN32
    {
        WSADATA wsa;
        WSAStartup(MAKEWORD(2, 2), &wsa);
    }
#else
    signal(SIGPIPE, SIG_IGN);
#endif
}

#ifdef _WIN32
static int32_t yar_net_status_from_error(int err) {
    switch (err) {
    case WSAECONNREFUSED:  return YAR_NET_REFUSED;
    case WSAETIMEDOUT:     return YAR_NET_TIMEOUT;
    case WSAEADDRINUSE:    return YAR_NET_ADDR_IN_USE;
    case WSAECONNRESET:
    case WSAECONNABORTED:  return YAR_NET_CONN_RESET;
    case WSAEACCES:        return YAR_NET_PERMISSION;
    case WSAEINVAL:
    case WSAENOTSOCK:
    case WSAEBADF:         return YAR_NET_INVALID_ARG;
    default:               return YAR_NET_IO;
    }
}
static int yar_net_last_error(void) { return WSAGetLastError(); }
#else
static int32_t yar_net_status_from_error(int err) {
    switch (err) {
    case ECONNREFUSED:     return YAR_NET_REFUSED;
    case ETIMEDOUT:        return YAR_NET_TIMEOUT;
    case EADDRINUSE:       return YAR_NET_ADDR_IN_USE;
    case ECONNRESET:
    case EPIPE:            return YAR_NET_CONN_RESET;
    case EACCES:           return YAR_NET_PERMISSION;
    case EINVAL:
    case EBADF:
    case ENOTSOCK:         return YAR_NET_INVALID_ARG;
    default:               return YAR_NET_IO;
    }
}
static int yar_net_last_error(void) { return errno; }
#endif

static void yar_net_close_socket(yar_sock_t sock) {
#ifdef _WIN32
    closesocket(sock);
#else
    close(sock);
#endif
}

static void yar_net_fill_addr(yar_net_addr *out, struct sockaddr_storage *sa) {
    char buf[INET6_ADDRSTRLEN];
    const char *host = NULL;
    int port = 0;

    if (sa->ss_family == AF_INET) {
        struct sockaddr_in *v4 = (struct sockaddr_in *)sa;
        host = inet_ntop(AF_INET, &v4->sin_addr, buf, sizeof(buf));
        port = ntohs(v4->sin_port);
    } else if (sa->ss_family == AF_INET6) {
        struct sockaddr_in6 *v6 = (struct sockaddr_in6 *)sa;
        host = inet_ntop(AF_INET6, &v6->sin6_addr, buf, sizeof(buf));
        port = ntohs(v6->sin6_port);
    }

    if (host != NULL) {
        out->host = yar_fs_copy_bytes(host, strlen(host));
    } else {
        out->host = yar_empty_str();
    }
    out->port = (int32_t)port;
}

static char *yar_net_port_str(int32_t port, char *buf, size_t bufsize) {
    snprintf(buf, bufsize, "%d", (int)port);
    return buf;
}

int32_t yar_net_listen(yar_str host, int32_t port, int64_t *out) {
    struct addrinfo hints, *res, *rp;
    yar_sock_t sock;
    int opt = 1;
    char portbuf[16];
    char *chost = NULL;
    int rc;

    *out = 0;
    yar_net_ensure_init();

    if (port < 0 || port > 65535) {
        return YAR_NET_INVALID_ARG;
    }

    /* Convert host to C string; empty host means all interfaces (NULL for getaddrinfo). */
    if (host.len > 0) {
        if (host.ptr == NULL) return YAR_NET_INVALID_ARG;
        chost = yar_fs_copy_cstr(host.ptr, (size_t)host.len);
    }

    memset(&hints, 0, sizeof(hints));
    hints.ai_family = AF_INET;
    hints.ai_socktype = SOCK_STREAM;
    hints.ai_flags = AI_PASSIVE;

    yar_net_port_str(port, portbuf, sizeof(portbuf));
    rc = getaddrinfo(chost, portbuf, &hints, &res);
    if (chost) free(chost);
    if (rc != 0) {
        return YAR_NET_NOT_FOUND;
    }

    sock = YAR_INVALID_SOCK;
    for (rp = res; rp != NULL; rp = rp->ai_next) {
        sock = socket(rp->ai_family, rp->ai_socktype, rp->ai_protocol);
        if (sock == YAR_INVALID_SOCK) continue;

        setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, (const char *)&opt, sizeof(opt));
#if defined(__APPLE__)
        {
            int nosig = 1;
            setsockopt(sock, SOL_SOCKET, SO_NOSIGPIPE, &nosig, sizeof(nosig));
        }
#endif

        if (bind(sock, rp->ai_addr, (int)rp->ai_addrlen) == 0) {
            break;
        }

        {
            int32_t status = yar_net_status_from_error(yar_net_last_error());
            yar_net_close_socket(sock);
            sock = YAR_INVALID_SOCK;
            if (rp->ai_next == NULL) {
                freeaddrinfo(res);
                return status;
            }
        }
    }
    freeaddrinfo(res);

    if (sock == YAR_INVALID_SOCK) {
        return YAR_NET_IO;
    }

    if (listen(sock, 128) != 0) {
        int32_t status = yar_net_status_from_error(yar_net_last_error());
        yar_net_close_socket(sock);
        return status;
    }

    *out = (int64_t)sock;
    return YAR_NET_OK;
}

int32_t yar_net_accept(int64_t listener, int64_t *out) {
    yar_sock_t lsock = (yar_sock_t)listener;
    yar_sock_t conn;
    struct sockaddr_storage addr;
    socklen_t addrlen = sizeof(addr);

    *out = 0;

    conn = accept(lsock, (struct sockaddr *)&addr, &addrlen);
    if (conn == YAR_INVALID_SOCK) {
        return yar_net_status_from_error(yar_net_last_error());
    }

#if defined(__APPLE__)
    {
        int nosig = 1;
        setsockopt(conn, SOL_SOCKET, SO_NOSIGPIPE, &nosig, sizeof(nosig));
    }
#endif

    *out = (int64_t)conn;
    return YAR_NET_OK;
}

int32_t yar_net_listener_addr(int64_t listener, yar_net_addr *out) {
    yar_sock_t sock = (yar_sock_t)listener;
    struct sockaddr_storage sa;
    socklen_t salen = sizeof(sa);

    out->host = yar_empty_str();
    out->port = 0;

    if (getsockname(sock, (struct sockaddr *)&sa, &salen) != 0) {
        return yar_net_status_from_error(yar_net_last_error());
    }

    yar_net_fill_addr(out, &sa);
    return YAR_NET_OK;
}

int32_t yar_net_close_listener(int64_t listener) {
    yar_sock_t sock = (yar_sock_t)listener;
    yar_net_close_socket(sock);
    return YAR_NET_OK;
}

int32_t yar_net_connect(yar_str host, int32_t port, int64_t *out) {
    struct addrinfo hints, *res, *rp;
    yar_sock_t sock;
    char portbuf[16];
    char *chost = NULL;
    int rc;
    int32_t status;

    *out = 0;
    yar_net_ensure_init();

    if (port < 0 || port > 65535) {
        return YAR_NET_INVALID_ARG;
    }
    if (host.len <= 0 || host.ptr == NULL) {
        return YAR_NET_INVALID_ARG;
    }

    chost = yar_fs_copy_cstr(host.ptr, (size_t)host.len);

    memset(&hints, 0, sizeof(hints));
    hints.ai_family = AF_INET;
    hints.ai_socktype = SOCK_STREAM;

    yar_net_port_str(port, portbuf, sizeof(portbuf));
    rc = getaddrinfo(chost, portbuf, &hints, &res);
    free(chost);
    if (rc != 0) {
        return YAR_NET_NOT_FOUND;
    }

    sock = YAR_INVALID_SOCK;
    status = YAR_NET_IO;
    for (rp = res; rp != NULL; rp = rp->ai_next) {
        sock = socket(rp->ai_family, rp->ai_socktype, rp->ai_protocol);
        if (sock == YAR_INVALID_SOCK) continue;

#if defined(__APPLE__)
        {
            int nosig = 1;
            setsockopt(sock, SOL_SOCKET, SO_NOSIGPIPE, &nosig, sizeof(nosig));
        }
#endif

        if (connect(sock, rp->ai_addr, (int)rp->ai_addrlen) == 0) {
            status = YAR_NET_OK;
            break;
        }

        status = yar_net_status_from_error(yar_net_last_error());
        yar_net_close_socket(sock);
        sock = YAR_INVALID_SOCK;
    }
    freeaddrinfo(res);

    if (sock == YAR_INVALID_SOCK) {
        return status;
    }

    *out = (int64_t)sock;
    return YAR_NET_OK;
}

int32_t yar_net_read(int64_t conn, int32_t max_bytes, yar_str *out) {
    yar_sock_t sock = (yar_sock_t)conn;
    char *buf;
    long long n;

    *out = yar_empty_str();

    if (max_bytes <= 0) {
        return YAR_NET_INVALID_ARG;
    }

    buf = (char *)malloc((size_t)max_bytes);
    if (buf == NULL) {
        return YAR_NET_IO;
    }

#ifdef _WIN32
    n = recv(sock, buf, max_bytes, 0);
#else
    n = recv(sock, buf, (size_t)max_bytes, 0);
#endif

    if (n < 0) {
        int32_t status = yar_net_status_from_error(yar_net_last_error());
        free(buf);
        return status;
    }

    if (n == 0) {
        /* EOF */
        free(buf);
        *out = yar_empty_str();
        return YAR_NET_OK;
    }

    *out = yar_fs_copy_bytes(buf, (size_t)n);
    free(buf);
    return YAR_NET_OK;
}

int32_t yar_net_write(int64_t conn, yar_str data, int32_t *out) {
    yar_sock_t sock = (yar_sock_t)conn;
    long long total = 0;
    int flags = 0;

    *out = 0;

    if (data.len <= 0) {
        return YAR_NET_OK;
    }
    if (data.ptr == NULL) {
        return YAR_NET_INVALID_ARG;
    }

#if defined(__linux__)
    flags = MSG_NOSIGNAL;
#endif

    while (total < data.len) {
        long long n;
#ifdef _WIN32
        n = send(sock, data.ptr + total, (int)(data.len - total), flags);
#else
        n = send(sock, data.ptr + total, (size_t)(data.len - total), flags);
#endif
        if (n <= 0) {
            if (total > 0) {
                *out = (int32_t)total;
                return YAR_NET_OK;
            }
            return yar_net_status_from_error(yar_net_last_error());
        }
        total += n;
    }

    *out = (int32_t)total;
    return YAR_NET_OK;
}

int32_t yar_net_close(int64_t conn) {
    yar_sock_t sock = (yar_sock_t)conn;
    yar_net_close_socket(sock);
    return YAR_NET_OK;
}

int32_t yar_net_set_read_deadline(int64_t conn, int32_t millis) {
    yar_sock_t sock = (yar_sock_t)conn;

    if (millis < 0) {
        return YAR_NET_INVALID_ARG;
    }

#ifdef _WIN32
    {
        DWORD tv = (DWORD)millis;
        if (setsockopt(sock, SOL_SOCKET, SO_RCVTIMEO, (const char *)&tv, sizeof(tv)) != 0) {
            return yar_net_status_from_error(yar_net_last_error());
        }
    }
#else
    {
        struct timeval tv;
        tv.tv_sec = millis / 1000;
        tv.tv_usec = (millis % 1000) * 1000;
        if (setsockopt(sock, SOL_SOCKET, SO_RCVTIMEO, &tv, sizeof(tv)) != 0) {
            return yar_net_status_from_error(yar_net_last_error());
        }
    }
#endif

    return YAR_NET_OK;
}

int32_t yar_net_resolve(yar_str host, int32_t port, yar_net_addr *out) {
    struct addrinfo hints, *res;
    char portbuf[16];
    char *chost = NULL;
    int rc;

    out->host = yar_empty_str();
    out->port = 0;

    yar_net_ensure_init();

    if (port < 0 || port > 65535) {
        return YAR_NET_INVALID_ARG;
    }
    if (host.len <= 0 || host.ptr == NULL) {
        return YAR_NET_INVALID_ARG;
    }

    chost = yar_fs_copy_cstr(host.ptr, (size_t)host.len);

    memset(&hints, 0, sizeof(hints));
    hints.ai_family = AF_INET;
    hints.ai_socktype = SOCK_STREAM;

    yar_net_port_str(port, portbuf, sizeof(portbuf));
    rc = getaddrinfo(chost, portbuf, &hints, &res);
    free(chost);
    if (rc != 0) {
        return YAR_NET_NOT_FOUND;
    }

    {
        struct sockaddr_storage sa;
        memcpy(&sa, res->ai_addr, res->ai_addrlen);
        yar_net_fill_addr(out, &sa);
    }

    freeaddrinfo(res);
    return YAR_NET_OK;
}
